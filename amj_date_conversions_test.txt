; ---------------------------------------------------------------------------
; Evaluate Current Date
; ---------------------------------------------------------------------------

set currentdate = format(curdate, "DD-MMM-YYYY;;D")
set currentday_dd = substring(1, 2, currentdate)
set currentmonth_mmm = substring(4, 3, currentdate)
set currentyear_yyyy = substring(8, 4, currentdate)

call echo(build2("currentday_dd = ", currentday_dd))
call echo(build2("currentmonth_mmm = ", currentmonth_mmm))
call echo(build2("currentyear_yyyy = ", currentyear_yyyy))
call echo(build2("currentdate = ", currentdate))

; Determine half-month run logic
if (cnvtint(currentday_dd) >= 16)
    set run_first_half = 1   ; Run for 1st half of current month
else
    set run_first_half = 0   ; Run for 2nd half of last month
endif

; Force to 2nd half for debug (optional)
set run_first_half = 0

; ---------------------------------------------------------------------------
; First Half of Current Month
; ---------------------------------------------------------------------------

if (run_first_half = 1)
    set first_half_start = cnvtdatetime(build2("01-", currentmonth_mmm, "-", currentyear_yyyy, " 00:00"))
    set first_half_end = cnvtdatetime(build2("15-", currentmonth_mmm, "-", currentyear_yyyy, " 23:59"))

    call echo(build2("first_half_start = ", format(first_half_start, "DD-MMM-YYYY HH:MM;;D")))
    call echo(build2("first_half_end   = ", format(first_half_end, "DD-MMM-YYYY HH:MM;;D")))

    set RUN_DATE_START = first_half_start
    set RUN_DATE_END = first_half_end
endif

; ---------------------------------------------------------------------------
; Second Half of Last Month
; ---------------------------------------------------------------------------

if (run_first_half = 0)
    set last_month_date_start = cnvtdatetime(datetimefind(cnvtlookbehind(curdate, "1,M"), "M", "B", "B"))
    set last_month_date_end = cnvtdatetime(datetimefind(cnvtlookbehind(curdate, "1,M"), "M", "E", "E"))

    call echo(build2("last_month_date_start = ", format(last_month_date_start, "DD-MMM-YYYY;;D")))
    call echo(build2("last_month_date_end   = ", format(last_month_date_end, "DD-MMM-YYYY;;D")))

    set lastmonthdate = format(last_month_date_end, "DD-MMM-YYYY;;D")
    set lastmonthday_dd = substring(1, 2, lastmonthdate)
    set lastmonthmonth_mmm = substring(4, 3, lastmonthdate)
    set lastmonthyear_yyyy = substring(8, 4, lastmonthdate)

    call echo(build2("lastmonthday_dd = ", lastmonthday_dd))
    call echo(build2("lastmonthmonth_mmm = ", lastmonthmonth_mmm))
    call echo(build2("lastmonthyear_yyyy = ", lastmonthyear_yyyy))
    call echo(build2("lastmonthdate = ", lastmonthdate))

    set second_half_start = cnvtdatetime(build2("16-", lastmonthmonth_mmm, "-", lastmonthyear_yyyy, " 00:00"))
    set second_half_end = cnvtdatetime(build2(lastmonthday_dd, "-", lastmonthmonth_mmm, "-", lastmonthyear_yyyy, " 23:59"))

    call echo(build2("second_half_start = ", format(second_half_start, "DD-MMM-YYYY HH:MM;;D")))
    call echo(build2("second_half_end   = ", format(second_half_end, "DD-MMM-YYYY HH:MM;;D")))

    set RUN_DATE_START = second_half_start
    set RUN_DATE_END = second_half_end
endif

; ---------------------------------------------------------------------------
; Final RUN DATES
; ---------------------------------------------------------------------------

set STARTRUN = format(RUN_DATE_START, "DD-MMM-YYYY HH:MM;;D")
set ENDRUN = format(RUN_DATE_END, "DD-MMM-YYYY HH:MM;;D")

call echo(build2("RUN_DATE_START = ", STARTRUN))
call echo(build2("RUN_DATE_END   = ", ENDRUN))

col 30 RUN_DATE_END
row+1

col 30 STARTRUN
col 60 "TO"
col 70 ENDRUN
row+1
with nocounter, maxcol = 200
