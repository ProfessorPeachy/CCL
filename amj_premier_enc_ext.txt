/*===============================================================================================================
TIMER LOGIC ALL AT ONCE WRITING AND LOGGING IN ONE SUBROUTINE
===============================================================================================================*/
DECLARE  log_time (( msg = vc )) =  VC  WITH  COPY
 
SUBROUTINE   log_time  ( msg  )
  set  section_timer->counter = section_timer->counter + 1
  set  stat = alterlist(section_timer->list, section_timer->counter)
  set  section_timer->list[section_timer->counter].section = msg
  set  section_timer->list[section_timer->counter].time = format(cnvtdatetime(curdate,curtime3), "@SHORTDATETIME")
  ;call echo(msg)
  ;Attempt to log out the data:
  if (section_timer->counter > 0)
   select into "ccluserdir:premier_timer_file.txt"
   from
	(dummyt d1 with seq = value(section_timer->counter))
  plan d1
   order by d1.seq
   head report
	delimit = "|" ,
	rep_delimit = "~" ,
	sub_delimit = "^"
	row + 1
   detail
	;Demos
	section_timer->list[d1.seq].section
	row + 1
	section_timer->list[d1.seq].time
	row + 1
	;cnvtstring(section_timer->list[d1.seq].qualifiers)
	row + 1
	"----------------------------------"
 
	row + 1
  foot report
	row + 2
 
  with  nocounter, maxcol = 2000
  ;with  nocounter , maxcol = 2000
  endif
RETURN ( msg ) END ;Subroutine

SUBROUTINE GET_CODE_VALUE_ALIAS(_CODE_VALUE)
	DECLARE _CV_ALIAS = VC
	declare passed_in_cv_numeric = f8
	declare cv_found = i2
	set cv_found = 0
	declare xNUM = i4
 
	if(isnumeric(_code_value)=1)
	   set passed_in_cv_numeric = cnvtreal(_CODE_VALUE)
	endif
 
 	IF (passed_in_cv_numeric > 0.0)
 
 	  set POS = locateval(xNUM, 1, code_val->count, passed_in_cv_numeric, code_val->qual[xNUM].cd)
 	  if(POS>0)
 	    ;if pos>0, found in record and return alias from record
 	    set _CV_ALIAS = code_val->qual[POS].alias
 	  else
 	    ;if not found in record, collect from database
  		SELECT into "nl:" ;029
  		CVA.ALIAS
  		FROM CODE_VALUE_OUTBOUND CVA
  		WHERE CVA.CODE_VALUE = passed_in_cv_numeric
  			AND CVA.CONTRIBUTOR_SOURCE_CD IN (HL7STDSRC_73_cv)
  		DETAIL
  			_CV_ALIAS = CVA.ALIAS
  			cv_found = 1
  			code_val->count = code_val->count + 1
  			if(code_val->size<code_val->count)
  			  code_val->size = code_val->size + 10
  			  stat = alterlist(code_val->qual,code_val->size)
  			endif
  			code_val->qual[code_val->count].cd = cva.code_value
  			code_val->qual[code_val->count].alias = cva.ALIAS
  			code_val->qual[code_val->count].disp = uar_get_code_display(passed_in_cv_numeric)
  		WITH TIME = 55
  		if(cv_found = 0);if no alias found, get the display for the value
  			set code_val->count = code_val->count + 1
  			if(code_val->size<code_val->count)
  			  set code_val->size = code_val->size + 10
  			  set stat = alterlist(code_val->qual,code_val->size)
  			endif
  			set code_val->qual[code_val->count].cd = passed_in_cv_numeric
  			set code_val->qual[code_val->count].alias = build("$$$",uar_get_code_display(passed_in_cv_numeric))
  			set code_val->qual[code_val->count].disp = build("$$$",uar_get_code_display(passed_in_cv_numeric))
;  			set code_val->qual[code_val->count].alias =uar_get_code_display(_CODE_VALUE)
;  			set code_val->qual[code_val->count].disp = uar_get_code_display(_CODE_VALUE)
  		endif
		endif
 
		IF (TRIM(_CV_ALIAS) > "")
			RETURN(_CV_ALIAS)
		ELSE
			RETURN(uar_get_code_display(passed_in_cv_numeric))
		ENDIF
	ELSE
		RETURN(_code_value)
	ENDIF
END ;SUBROUTINE
